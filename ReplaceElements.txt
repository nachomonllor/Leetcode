 public static int[] ReplaceElements(int[] arr)
        {
            int max = arr.Last();
            for (int i = arr.Length - 1; i >= 0; i--)
            {
                int aux = arr[i];
                arr[i] = max;
                max = Math.Max(max, aux);
            }
            arr[arr.Length - 1] = -1;
            return arr;
        }
